class Quest:
    def __init__(self, id, name, prereqs, objectives, rewards):
        self.id = id
        self.name = name
        self.prereqs = prereqs  # {'level': int, 'previous_quests': [int], 'items': [Item]}
        self.objectives = objectives  # [{'type': str, 'value': int}]
        self.rewards = rewards  # {'experience': int, 'items': [Item]}

class Player:
    def __init__(self, id):

2. Fluxo de Progressão:

```python
def complete_objective(player, quest_id, objective_type, value):
    for quest in player.quests:
        if quest.id == quest_id:
            for i, obj in enumerate(quest.objectives):
                if obj['type'] == objective_type and obj['value'] == value:
                    # Marca como concluído
                    quest.objectives[i]['completed'] = True
                    return True
    return False

def check_prereqs(player, quest):
    if quest.prereqs.get('level') and player.level < quest.prereqs['level']:
        return False
    for q_id in quest.prereqs.get('previous_quests', []):
        if q_id not in player.completed_quests:
            return False
    for item in quest.prereqs.get('items', []):
        if item.id not in player.inventory or player.inventory[item.id] < item.quantity:
            return False
    return True

def complete_quest(player, quest_id):
    for i, quest in enumerate(player.quests):
        if quest.id == quest_id and all(obj['completed'] for obj in quest.objectives):
            # Marca como concluído
            player.completed_quests.append(quest_id)
            player.quests.pop(i)
            return True
    return False
```

3. Validação Eficiente:

A validação de requisitos é feita em tempo linear com relação ao número de requisitos. Usamos listas e dicionários para armazenar informações, o que permite acessos rápidos.

4. Exemplo Concreto:

```python
# Questa quest requer level 10, ter completado a quest 1 e coletar 5 itens de ID 101
quest = Quest(2, "Quest Example", {'level': 10, 'previous_quests': [1], 'items': [{'id': 101, 'quantity': 5}]}, 
              [{'type': 'collect', 'value': 5}], {'experience': 100})

# Player começa com level 12 e item ID 101
player = Player(1)
player.level = 12
player.inventory = {101: 3}

# Completa a quest
complete_objective(player, 2, 'collect', 5)  # Marca como concluído
complete_quest(player, 2)  # Quest completa e adicionada ao histórico

print(f"Player completed quests: {player.completed_quests}")
```

Este exemplo mostra como uma quest é implementada com a estrutura de dados definida. O jogador pode completar partes da quest e, quando todas as partes forem concluídas, a quest inteira é marcada como completa e adicionada ao histórico do jogador.
