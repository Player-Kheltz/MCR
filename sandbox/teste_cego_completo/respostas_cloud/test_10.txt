## Fragmentador Refatorado - 4 responsabilidades separadas

```python
"""Modulo: Fragmentador - Pipeline de geracao de codigo sob medida."""
import os, re, math, json, urllib.request

BASE = os.path.abspath(os.path.join(os.path.dirname(__file__), '..', '..', '..'))
SANDBOX = os.path.join(BASE, 'sandbox')
OLLAMA_URL = os.environ.get('OLLAMA_URL', 'http://localhost:11434/api/generate')

def init_module(contexto):
    return 'fragmentador', Fragmentador()

class Fragmentador:
    """Pipeline de geracao de codigo com fragmentos dinamicos."""
    
    def executar(self, descricao):
        """Fluxo principal: extrair -> decidir -> gerar -> salvar."""
        requisitos = self._extrair_requisitos(descricao)
        if not requisitos:
            print('[Fragmentador] Nao foi possivel extrair requisitos')
            return None
        
        estrategia = self._decidir_estrategia(requisitos)
        print(f'[Fragmentador] {requisitos["nome"]} | ~{requisitos["linhas_est"]} linhas | {estrategia["num_frag"]} fragmento(s)')
        
        codigo = self._gerar_codigo(descricao, estrategia)
        if not codigo:
            print('[Fragmentador] Falha ao gerar codigo')
            return None
        
        return self._salvar_arquivo(codigo, requisitos["nome"])
    
    def _extrair_requisitos(self, descricao):
        """(1) Extrai nome do arquivo e estimativa de tamanho da descricao."""
        nome = self._extrair_nome(descricao)
        linhas_est = self._detectar_tamanho(descricao)
        return {"nome": nome, "descricao": descricao, "linhas_est": linhas_est}
    
    def _decidir_estrategia(self, requisitos):
        """(2) Decide se gera codigo unico ou fragmentado."""
        linhas = requisitos["linhas_est"]
        num_frag = max(1, math.ceil(linhas / 80))
        return {"unico": num_frag == 1, "num_frag": num_frag, "tam_frag": 80}
    
    def _gerar_codigo(self, descricao, estrategia):
        """(3) Gera o codigo via IA, unico ou fragmentado."""
        if estrategia["unico"]:
            return self._gerar_direto(descricao)
        return self._gerar_fragmentado(descricao, estrategia["num_frag"])
    
    def _salvar_arquivo(self, codigo, nome_arquivo):
        """(4) Salva o codigo em arquivo e retorna o caminho."""
        caminho = os.path.join(SANDBOX, nome_arquivo)
        with open(caminho, 'w', encoding='utf-8') as f:
            f.write(codigo)
        print(f'[Fragmentador] OK: {caminho} ({len(codigo.splitlines())} linhas)')
        return caminho
    
    # --- Metodos auxiliares (extraidos do original) ---
    
    def _extrair_nome(self, desc):
        nome = re.search(r'(?:chamado|criar|em|como)\s+([\w\-]+\.(?:py|md|json|xml|lua|txt))', desc.lower())
        if nome: return nome.group(1)
        nome = re.search(r'(\w+\.py)', desc.lower())
        if nome: return nome.group(1)
        return f'script_{hash(desc) % 10000}.py'
    
    def _detectar_tamanho(self, desc):
        desc_lower = desc.lower()
        if any(p in desc_lower for p in ['sistema', 'framework', 'modulo completo', 'motor', 'engine']):
            return 300
        if any(p in desc_lower for p in ['classe', 'funcao', 'modulo', 'utilitario', 'ferramenta']):
            return 80
        if any(p in desc_lower for p in ['script', 'exemplo', 'teste', 'hello world', 'pequeno']):
            return 15
        return max(10, min(200, len(desc) // 20))
    
    def _gerar_direto(self, desc):
        prompt = f"Gere codigo Python completo para:\n{desc}\n\nRetorne APENAS o codigo em ```python ... ```"
        resp = self._ia(prompt, 0.3)
        m = re.search(r'```(?:python)?\s*\n(.+?)```', resp, re.DOTALL)
        return m.group(1).strip() if m else re.sub(r'```\w*\n?', '', resp).strip()
    
    def _gerar_fragmentado(self, desc, num_frag):
        fragmentos = []
        for i in range(num_frag):
            ctx = '\n'.join(fragmentos) if fragmentos else ''
            prompt = (
                f"Gere codigo Python para:\n{desc}\n\n"
                f"Bloco {i+1}/{num_frag}.\n"
                f"{'Contexto:\n' + ctx if ctx else ''}\n"
                f"Retorne em ```python ... ```"
            )
            resp = self._ia(prompt, 0.3)
            m = re.search(r'```(?:python)?\s*\n(.+?)```', resp, re.DOTALL)
            bloco = m.group(1).strip() if m else re.sub(r'```\w*\n?', '', resp).strip()
            if bloco: fragmentos.append(bloco)
        return '\n\n'.join(fragmentos) if fragmentos else None
    
    def _ia(self, prompt, temp=0.3):
        d = json.dumps({'model': 'qwen2.5-coder:7b', 'prompt': prompt, 'stream': False,
            'options': {'temperature': temp, 'num_ctx': 4096}}).encode()
        r = urllib.request.Request(OLLAMA_URL, data=d, headers={'Content-Type': 'application/json'})
        return json.loads(urllib.request.urlopen(r, timeout=120).read()).get('response', '')
```

### Principais mudancas:
1. `executar()` agora orquestra 4 passos, cada um delegado a um metodo especifico
2. `_extrair_requisitos()` unifica extracao de nome e tamanho
3. `_decidir_estrategia()` isola a logica de quantos fragmentos usar
4. `_gerar_codigo()` faz o roteamento entre direto e fragmentado
5. `_salvar_arquivo()` lida exclusivamente com I/O
6. Cada metodo tem UMA responsabilidade clara e testavel isoladamente
7. A interface publica (`executar()`) permanece identica - 100% compativel
