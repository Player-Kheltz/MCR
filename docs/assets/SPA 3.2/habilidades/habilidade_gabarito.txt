--[[
    Projeto MCR – Habilidade Gabarito (v1.0)
    =============================================
    Este ficheiro documenta TODOS os campos que uma habilidade do SPA pode ter.
    Use‑o como referência ao criar novas habilidades nos ficheiros modulares.
    NÃO contém lógica executável — é apenas uma referência comentada.

    Campos organizados por:
    [O] = Obrigatório
    [C] = Condicional (obrigatório apenas para certos tipos)
    [OP] = Opcional
--]]

-- ============================================================
-- EXEMPLO DE UMA HABILIDADE COMPLETA (com todos os campos)
-- ============================================================

--[[
HABILIDADES[99] = {
    -- =====================
    -- CAMPOS OBRIGATÓRIOS
    -- =====================

    -- [O] Nome visível no jogo (emote sobre o jogador)
    nome = "Nome da Habilidade",

    -- [O] Tipo da habilidade:
    --     "passiva"        – efeito permanente (aplicado no login e ao subir de nível)
    --     "gatilho"        – ativada por eventos (ataque, defesa, kill, etc.)
    --     "transformacao"  – activa/desactiva com base nos estados de alma (Lampejo/Vínculo)
    tipo = "gatilho",

    -- [O] Lista de IDs de domínios onde esta habilidade se aplica.
    --     Ex.: {100} = só Espadas Leves; {110, 111} = Machados Leves e Pesados.
    dominio = {100},

    -- [O] Nível mínimo do domínio para aprender a habilidade automaticamente.
    --     Se aprendida antes (por quest/NPC), funciona mas com valores de domínio baixo.
    nivelMin = 5,

    -- [O] Descrição curta do efeito (aparece no chat ao aprender e ao ativar)
    descricaoEfeito = "causa sangramento no alvo",

    -- [O] Cor da habilidade no chat (da tabela COR)
    cor = COR.DOM_COMBATE_LAMINAS,

    -- [O] Mensagens de feedback para cada postura.
    --     [1] = Ímpeto, [2] = Equilíbrio, [3] = Guarda.
    --     Se não quiseres mensagens diferentes, repete a mesma string.
    mensagem = {
        [1] = "Mensagem para Ímpeto!",
        [2] = "Mensagem para Equilíbrio.",
        [3] = "Mensagem para Guarda."
    },

    -- =====================
    -- CAMPOS CONDICIONAIS
    -- =====================

    -- [C] Só para tipo "gatilho". Define o evento que activa a habilidade:
    --     "onAttack"         – ao causar dano
    --     "onReceiveDamage"  – ao receber dano
    --     "onKill"           – ao matar uma criatura
    --     "onBlock"          – ao bloquear um ataque (com escudo)
    --     "onHeal"           – ao curar alguém
    gatilho = "onAttack",

    -- [C] Só para tipo "gatilho". Probabilidade base (0–100).
    chanceBase = 12,

    -- [C] Só para tipo "gatilho". Função que executa o efeito principal.
    --     Parâmetros: (player, target, damage)
    --     Para onKill: (player, target) – sem damage
    --     Para onHeal: (player, target, healAmount)
    --     Para onReceiveDamage/onBlock: (player, attacker, damage)
    efeito = function(player, target, damage)
        -- Lógica da habilidade
        -- Pode usar getNivelPorAfinidade() para escalar com o domínio
        -- Pode usar player:getFightMode() para ajustar por postura
    end,

    -- [C] Só para tipo "transformacao". Função que activa/desactiva a transformação.
    --     Parâmetros: (player)
    --     Chamada quando o estado de alma (Lampejo/Vínculo) muda.
    --     Exemplo: player:setStorageValue(XXXXX, 1)
    -- efeito = function(player) ... end,

    -- =====================
    -- CAMPOS OPCIONAIS
    -- =====================

    -- [OP] Só para tipo "gatilho". Se true, o lock (STORAGE.HABILIDADE_LOCK)
    --      é gerido manualmente dentro da função efeito (útil para addEvent).
    async = true,

    -- [OP] Só para tipo "gatilho". Tipo de buff a aplicar (alternativa a custom).
    --      Valores possíveis:
    --      "speedBurst"          – acelera velocidade de ataque
    --      "nextAttackCondition" – aplica condição no próximo ataque
    --      "damageBoost"         – aumenta dano por N ataques
    --      "lifeLeech"           – roubo de vida por duração
    --      "custom"              – efeito totalmente personalizado (usa o campo 'efeito')
    buffType = "custom",

    -- [OP] Parâmetros para speedBurst (obrigatórios se buffType = "speedBurst")
    buffMultiplier = 0.1,     -- multiplicador da velocidade (0.1 = 10x mais rápido)
    buffDurationMs = 500,     -- duração do burst em milissegundos

    -- [OP] Parâmetros para nextAttackCondition
    conditionType = CONDITION_BLEEDING,   -- tipo de condição
    conditionDuration = 4000,             -- duração em ms
    conditionDamage = 20,                 -- dano por tick (se aplicável)

    -- [OP] Parâmetros para damageBoost
    percentExtra = 20,        -- percentagem de dano extra
    buffHits = 3,             -- número de ataques com o bónus

    -- [OP] Parâmetros para lifeLeech
    leechPercent = 15,        -- percentagem de life leech
    durationSec = 5,          -- duração em segundos

    -- [OP] Probabilidade durante Lampejo. Se omitido, usa chanceBase.
    chanceLampejo = 22,

    -- [OP] Probabilidade durante Vínculo. Se omitido, usa chanceBase.
    chanceVinculo = 30,

    -- [OP] Condição especial para ativar a habilidade:
    --      "cercado"     – 3+ inimigos ao redor
    --      "vidaBaixa"   – alvo com menos de 20% de vida
    --      "distancia4"  – alvo a 4+ tiles de distância
    --      "5Golpes"     – após 5 ataques consecutivos com a mesma arma
    condicao = "cercado",

    -- [OP] Função para efeitos visuais (projécteis, magias no alvo).
    --      Executada sempre que a habilidade é activada.
    efeitoVisual = function(player, target)
        sendDistanceEffect(player, target, CONST_ANI_ARROW, CONST_ME_DRAWBLOOD)
    end,

    -- [OP] Modificadores por postura.
    --      chanceMult: multiplicador da chance base (1.0 = normal, 1.5 = +50%)
    --      Podes adicionar outros campos personalizados que o teu efeito leia.
    posturaMod = {
        [1] = { chanceMult = 1.3 },   -- Ímpeto: +30% de chance
        [2] = { chanceMult = 1.0 },   -- Equilíbrio: normal
        [3] = { chanceMult = 0.5 },   -- Guarda: -50% de chance
    },
}
--]]

-- ============================================================
-- NOTAS IMPORTANTES
-- ============================================================

--[[
    1. EVOLUÇÃO COM O DOMÍNIO:
       Dentro da função 'efeito', podes usar:
         local nivel = getNivelPorAfinidade(player:getDominioAfinidade(DOMINIO_ID))
       para ajustar dano, duração, número de golpes, etc.

    2. POSTURA:
       Dentro da função 'efeito', podes usar:
         local postura = player:getFightMode()
       para alterar completamente o comportamento (ex.: mais golpes em Ímpeto).

    3. HABILIDADES APRENDIDAS ANTES DO NÍVEL:
       Se a habilidade for desbloqueada por quest/NPC antes do nível recomendado,
       o 'efeito' deve verificar o nível do domínio e comportar-se de acordo
       (valores base para domínio baixo, valores épicos para Maestria).

    4. LOCKS ASSÍNCRONOS:
       Se usares 'addEvent' dentro do 'efeito', define async = true
       e gere o lock manualmente:
         player:setStorageValue(STORAGE.HABILIDADE_LOCK + habId, 1)  -- antes
         player:setStorageValue(STORAGE.HABILIDADE_LOCK + habId, 0)  -- depois

    5. BUFFS PRÉ-DEFINIDOS:
       Se usares buffType = "speedBurst" / "nextAttackCondition" / etc.,
       NÃO uses o campo 'efeito'. O executor trata desses buffs automaticamente.
--]]
