--[[
    Projeto MCR – SPA – Habilidade Gabarito (v2.2)
    =============================================
    Documentação completa de TODOS os campos que uma habilidade do SPA pode ter.
    Atualizado com:
      - Novas condições ("cercadoAlvo")
      - Uso de atributos de itens (getAttack, getDefense, etc.)
      - Projéteis dinâmicos via getProjectile()
      - Executor inteligente (prioridade, cooldown, categoria)
      - Combos automáticos gerenciados pelo executor
      - Escalonamento por nível de domínio e postura
      - Árvore I‑IV e ensino entre domínios irmãos
    Campos organizados por:
    [O]  = Obrigatório
    [C]  = Condicional (obrigatório apenas para certos tipos)
    [OP] = Opcional
--]]

-- ============================================================
-- EXEMPLO DE UMA HABILIDADE COMPLETA (todos os campos possíveis)
-- ============================================================

--[[
HABILIDADES[99999] = {
    -- =====================
    -- CAMPOS OBRIGATÓRIOS
    -- =====================

    -- [O] Nome visível no jogo (emote sobre o jogador e mensagens)
    nome = "Nome da Habilidade",

    -- [O] Tipo da habilidade:
    --     "passiva"        – efeito permanente (aplicado no login e ao subir de nível)
    --     "gatilho"        – ativada por eventos (ataque, defesa, kill, etc.)
    --     "transformacao"  – activa/desactiva com base nos Estados de Alma (Lampejo/Vínculo)
    tipo = "gatilho",

    -- [O] Lista de IDs de domínios onde esta habilidade se aplica.
    --     Ex.: {100} = só Espadas Leves; {110, 111} = Machados Leves e Pesados.
    dominio = {100},

    -- [O] Nível mínimo do domínio para aprender a habilidade automaticamente.
    --     Se aprendida antes (por quest/NPC), funciona mas com valores de domínio baixo.
    nivelMin = 5,

    -- [O] Descrição curta do efeito (aparece no chat ao aprender e ao ativar).
    --     Pode ser uma string fixa ou uma função que recebe (player) e retorna uma string.
    descricaoEfeito = "causa dano adicional",

    -- [O] Cor da habilidade no chat (da tabela COR). Use as constantes COR.DOM_* ou COR.POSTURA_*.
    cor = COR.DOM_COMBATE_LAMINAS,

    -- [O] Mensagens de feedback para cada postura.
    --     [1] = Ímpeto, [2] = Equilíbrio, [3] = Guarda.
    --     O executor usará a mensagem da postura atual; se não existir, usa [2].
    mensagem = {
        [1] = "Mensagem para Ímpeto!",
        [2] = "Mensagem para Equilíbrio.",
        [3] = "Mensagem para Guarda."
    },

    -- =====================
    -- CAMPOS CONDICIONAIS (GATILHO)
    -- =====================

    -- [C] Só para tipo "gatilho". Define o evento que activa a habilidade:
    --     "onAttack"         – ao causar dano
    --     "onDealtDamage"    – após causar dano (disparado pelo eventos.lua)
    --     "onKill"           – ao matar uma criatura
    --     "onBlock"          – ao bloquear um ataque (com escudo)
    --     "onReceiveDamage"  – ao receber dano
    --     "onHeal"           – ao curar alguém (futuro)
    gatilho = "onAttack",

    -- [C] Só para tipo "gatilho". Cooldown individual em segundos.
    --     Após usar, a habilidade fica indisponível por este período.
    cooldown = 8,

    -- [C] Só para tipo "gatilho". Prioridade base (1‑10).
    --     Habilidades com maior prioridade são escolhidas primeiro pelo motor inteligente.
    prioridade = 5,

    -- [C] Só para tipo "gatilho". Categoria da habilidade:
    --     "single"        – dano direto elevado
    --     "aoe"           – atinge múltiplos alvos
    --     "debuff"        – aplica condição negativa (sangramento, atordoamento, lentidão)
    --     "buff"          – melhora o próprio jogador (speed burst, dano extra)
    --     "defense"       – cura, escudo, redução de dano
    --     "chi_spender"   – consome Chi (requer chiCusto definido)
    --     "finisher"      – efeito extra quando alvo está com vida baixa
    categoria = "single",

    -- [C] Só para tipo "gatilho". Função que executa o efeito principal.
    --     Parâmetros dependem do gatilho:
    --       onAttack, onDealtDamage: (player, target, damage)
    --       onKill: (player, target)
    --       onBlock, onReceiveDamage: (player, attacker, damage)
    --     Dentro da função, podes usar:
    --       getNivelPorAfinidade() para escalar com o domínio
    --       player:getFightMode() para ajustar por postura
    --       item:getAttack(), item:getDefense(), etc. para escalar com o equipamento
    efeito = function(player, target, damage)
        -- Lógica da habilidade
    end,

    -- [C] Só para tipo "transformacao". Função que activa/desactiva a transformação.
    --     Parâmetros: (player)
    --     Chamada quando o estado de alma (Lampejo/Vínculo) muda.
    -- efeito = function(player) ... end,

    -- =====================
    -- CAMPOS OPCIONAIS (GATILHO)
    -- =====================

    -- [OP] Se true, o lock (STORAGE.HABILIDADE_LOCK) é gerido manualmente dentro da função efeito.
    --      Útil para addEvent assíncronos.
    async = false,

    -- [OP] Tipo de buff automático a aplicar (alternativa a custom).
    --      Valores possíveis:
    --      "speedBurst"          – acelera velocidade de ataque
    --      "nextAttackCondition" – aplica condição no próximo ataque
    --      "damageBoost"         – aumenta dano por N ataques
    --      "lifeLeech"           – roubo de vida por duração
    --      "custom"              – efeito totalmente personalizado (usa o campo 'efeito')
    buffType = "custom",

    -- [OP] Parâmetros para speedBurst
    buffMultiplier = 0.4,     -- multiplicador da velocidade (0.4 = 2.5x mais rápido)
    buffDurationMs = 1200,    -- duração em milissegundos

    -- [OP] Parâmetros para nextAttackCondition
    conditionType = CONDITION_BLEEDING,
    conditionDuration = 4000,
    conditionDamage = 15,

    -- [OP] Parâmetros para damageBoost
    percentExtra = 20,        -- percentagem de dano extra
    buffHits = 3,             -- número de ataques com o bónus

    -- [OP] Parâmetros para lifeLeech
    leechPercent = 15,        -- percentagem de life leech
    durationSec = 5,          -- duração em segundos

    -- [OP] Probabilidade durante Lampejo (%). Se omitido, usa a prioridade base.
    chanceLampejo = 22,

    -- [OP] Probabilidade durante Vínculo (%). Se omitido, usa a prioridade base.
    chanceVinculo = 30,

    -- [OP] (OBSOLETO) chanceBase era usado no sistema antigo. Agora o motor inteligente
    --      ignora este campo. Use 'prioridade' e 'cooldown' em vez disso.
    -- chanceBase = 12,

    -- [OP] Condição especial para activar a habilidade:
    --      "cercado"      – 3+ inimigos ao redor do jogador
    --      "cercadoAlvo"  – 2+ inimigos perto do alvo (raio 2 tiles)
    --      "vidaBaixa"    – alvo com menos de 20% de vida
    --      "distancia4"   – alvo a 4+ tiles de distância
    --      "5Golpes"      – após 5 ataques consecutivos com a mesma arma
    condicao = "cercado",

    -- [OP] Custo de Chi (para habilidades que consomem Chi).
    --      Só funciona se o domínio tiver acesso ao sistema de Chi.
    chiCusto = 5,

    -- [OP] Função para efeitos visuais (projécteis, magias no alvo).
    --      Executada sempre que a habilidade é activada.
    efeitoVisual = function(player, target)
        -- Para projéteis (armas de distância), usar getProjectile(player) para obter
        -- a animação correta com base na munição real.
        local proj = getProjectile(player)
        if proj then
            sendDistanceEffect(player, target, proj, CONST_ME_DRAWBLOOD)
        else
            -- Para corpo‑a‑corpo, usar efeitos mágicos no alvo
            target:getPosition():sendMagicEffect(CONST_ME_DRAWBLOOD)
        end
    end,

    -- [OP] Modificadores por postura.
    --      chanceMult: multiplicador da prioridade na postura (1.0 = normal, 1.5 = +50%)
    --      Podes adicionar outros campos personalizados que o teu efeito leia.
    posturaMod = {
        [1] = { chanceMult = 1.3 },   -- Ímpeto
        [2] = { chanceMult = 1.0 },   -- Equilíbrio
        [3] = { chanceMult = 0.7 },   -- Guarda
    },

    -- =====================
    -- EVOLUÇÃO DA HABILIDADE COM O DOMÍNIO E POSTURA
    -- =====================

    --[[
    As habilidades podem desbloquear efeitos adicionais com base no nível de
    afinidade do domínio e/ou na postura actual. Isto é implementado dentro
    da função 'efeito' com verificações simples:

    Exemplo de evolução por nível de domínio:
        local afinidade = player:getDominioAfinidade(DOMINIO_ID) or 0
        local nivel = getNivelPorAfinidade(afinidade)

        -- Efeito base (sempre presente)
        causarDanoExtra(player, target, danoBase)

        -- Nível 10: adiciona sangramento
        if nivel >= 10 then
            aplicarSangramento(target, duracao, danoPorTick)
        end

        -- Nível 15: adiciona stun
        if nivel >= 15 then
            aplicarStun(target, duracao)
        end

        -- Nível 20 (Maestria): efeito extra massivo
        if nivel >= 20 then
            causarDanoEmArea(player, target, danoMassivo)
        end

    Exemplo de evolução por postura + nível:
        local postura = player:getFightMode()
        local nivel = getNivelPorAfinidade(afinidade)

        if postura == FIGHTMODE_ATTACK and nivel >= 10 then
            -- Em Ímpeto com nível 10+, a habilidade causa knockback adicional
            knockback(target)
        end

    Podes combinar ambas as condições para criar ramificações complexas.
    --]],

    -- =====================
    -- REGRAS GERAIS PARA TODAS AS HABILIDADES
    -- =====================

    --[[
    1. ESCALONAMENTO COM O DOMÍNIO:
       Dentro da função 'efeito', usa:
         local nivel = getNivelPorAfinidade(player:getDominioAfinidade(DOMINIO_ID))
       para ajustar dano, duração, número de golpes, E TAMBÉM DESBLOQUEAR
       NOVOS EFEITOS ao atingir certos níveis (ver secção EVOLUÇÃO acima).
       Exemplo: dano extra = danoBase * (0.5 + nivel * 0.1)

    2. INFLUÊNCIA DA POSTURA:
       Dentro da função 'efeito', podes usar:
         local postura = player:getFightMode()
       para alterar completamente o comportamento. As mensagens já são
       adaptadas automaticamente. Efeitos específicos podem ser desbloqueados
       apenas em determinadas posturas (ex.: stun só em Guarda).

    3. USO DE ATRIBUTOS DE ITENS:
       Agora que temos acesso a item:getAttack(), item:getDefense(),
       item:getArmor(), item:getSkillFist(), etc., as habilidades devem
       preferencialmente escalar com base nos equipamentos reais.
       Exemplo: dano extra = (item:getAttack() or 0) * 0.5.
       Isto torna cada arma única e recompensa o jogador por conseguir
       equipamentos melhores.

    4. HABILIDADES APRENDIDAS ANTES DO NÍVEL:
       Se a habilidade for desbloqueada por quest/NPC antes do nível recomendado,
       o 'efeito' deve verificar o nível do domínio e comportar-se de acordo
       (valores base para domínio baixo, valores épicos para Maestria).

    5. ÁRVORE I‑IV OBRIGATÓRIA:
       Toda especialidade deve ter 4 habilidades principais:
         I   (nível 5):  Passiva que define a identidade do estilo.
         II  (nível 10): Passiva que expande a primeira.
         III (nível 15): Gatilho ou Transformação — a "assinatura" do estilo.
         IV  (nível 20): Maestria — Passiva que concede Serenidade e/ou
                         aperfeiçoa a habilidade III e ensina‑a ao domínio irmão.

    6. ENSINO ENTRE DOMÍNIOS IRMÃOS:
       Domínios irmãos (ex.: Espadas Leves e Pesadas; Arcos e Arremesso;
       Fogo/Ar e Água/Gelo; Primários Combate e Magia) podem ensinar habilidades
       um ao outro no nível 20. Implementar via storage ou diretamente na função
       de Maestria.

    7. PROJÉTEIS (HABILIDADES DE DISTÂNCIA):
       Para habilidades que disparam múltiplos projéteis, segue o padrão:
         a. Obtém o projétil real UMA VEZ antes dos addEvent, usando getProjectile(player).
         b. Dispara cada projétil em addEvent separados, com intervalo MÍNIMO
            de 250ms (NUNCA MENOS QUE ISSO). O intervalo exato depende da
            habilidade e deve ser ajustado para o melhor feedback visual.
         c. Aplica o dano ANTES ou ao mesmo tempo que o projétil.

    8. HABILIDADES CORPO‑A‑CORPO:
       Para golpes múltiplos, usa efeitos visuais no alvo (CONST_ME_HITAREA,
       CONST_ME_DRAWBLOOD) e, se apropriado, knockback. O feedback de dano
       será naturalmente visível porque os danos são aplicados em eventos
       separados com intervalos >= 250ms.

    9. FEEDBACK NARRATIVO:
       Toda habilidade deve ter mensagens claras e imersivas. Usa cores da
       tabela COR para destacar o nome da habilidade e o efeito.
       Exemplo: c("[Disparo Triplo] ", COR.COMBO_AVISO) .. "Três flechas!"

    10. BUFFS E DEBUFFS:
        Habilidades que aplicam condições (sangramento, atordoamento, etc.)
        devem usar o buffType "nextAttackCondition" ou aplicar a condição
        diretamente no efeito customizado.

    11. CHI:
        Habilidades que consomem Chi devem definir chiCusto e verificar o
        saldo antes de executar. O motor inteligente já faz essa verificação.
        A ponte CHI_ACCUMULATED → HARMONY será implementada num think próprio.

    12. COMBOS AUTOMÁTICOS:
        O executor gere sequências de até 3 habilidades consecutivas com base
        no nível de maestria do domínio. As habilidades não precisam de
        configuração extra para participar dos combos — basta estarem elegíveis.
        Os campos cooldown e prioridade controlam a frequência com que cada
        habilidade aparece nos combos.
    --]],
}

    if proj then
        sendDistanceEffect(player, target, proj, CONST_ME_DRAWBLOOD)
    end
